    private static void ordenamientoArreglo(){
        int []arr = {5,2,3,1,4};
        int temp;
        
        for (int i = 1; i < arr.length; i++) {
            for (int j = arr.length-1; j >= i; j--) {
                if(arr[j] < arr[j-1]){
                    temp=arr[j];
                    arr[j] = arr[j-1];
                    arr[j-1] = temp;
                }
            }
        }
        
        for (int i = 0; i < arr.length; i++) {
            System.out.println(arr[i]);
        }
    }
